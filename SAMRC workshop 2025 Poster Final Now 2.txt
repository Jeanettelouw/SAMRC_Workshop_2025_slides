    <script>
        const transitionCard = document.getElementById('transitionCard');
        const reactiveText = document.getElementById('reactiveText');
        const proactiveText = document.getElementById('proactiveText');
        let isProactive = false;

        transitionCard.addEventListener('click', function() {
            if (isProactive) {
                // Transition from Proactive to Reactive
                transitionCard.classList.add('transitioning');
                proactiveText.classList.remove('fade-in');
                proactiveText.classList.add('fade-out');
                
                setTimeout(() => {
                    proactiveText.classList.remove('fade-out');
                    proactiveText.classList.add('hidden');
                    reactiveText.classList.remove('hidden');
                    reactiveText.classList.add('fade-in');
                }, 500);

                setTimeout(() => {
                    transitionCard.classList.remove('transitioning');
                    transitionCard.style.background = 'linear-gradient(135deg, #e74c3c 0%, #c0392b 100%)';
                }, 1000);
            } else {
                // Transition from Reactive to Proactive
                transitionCard.classList.add('transitioning');
                reactiveText.classList.remove('fade-in');
                reactiveText.classList.add('fade-out');
                
                setTimeout(() => {
                    reactiveText.classList.remove('fade-out');
                    reactiveText.classList.add('hidden');
                    proactiveText.classList.remove('hidden');
                    proactiveText.classList.add('fade-in');
                }, 500);

                setTimeout(() => {
                    transitionCard.classList.remove('transitioning');
                    transitionCard.style.background = 'linear-gradient(135deg, #27ae60 0%, #229954 100%)';
                }, 1000);
            }
            
            isProactive = !isProactive;
        });
    </script>
